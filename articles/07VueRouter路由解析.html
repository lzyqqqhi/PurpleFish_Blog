              # 2. 插件初始化 install 关键差异

              # Vue2 Router

              // src/install.js
export function install (Vue) {
  if (install.installed) return
  install.installed = true
  _Vue = Vue

  Vue.mixin({
    beforeCreate () {
      if (this.$options.router) {
        this._routerRoot = this
        this._router = this.$options.router
        this._router.init(this)
        // 响应式 _route
        Vue.util.defineReactive(this, '_route', this._router.history.current)
      } else {
        this._routerRoot = (this.$parent && this.$parent._routerRoot) || this
      }
    }
  })

  // 原型链注入
  Object.defineProperty(Vue.prototype, '$router', {
    get () { return this._routerRoot._router }
  })
  Object.defineProperty(Vue.prototype, '$route', {
    get () { return this._routerRoot._route }
  })

  // 全局组件
  Vue.component('RouterView', View)
  Vue.component('RouterLink', Link)
}

              # Vue3 Router

              // src/install.ts
export const install = (app: App) => {
  const installed = !!(app as any)[routerKey]
  if (installed) return
  ;(app as any)[routerKey] = true

  app.mixin({
    beforeCreate(this: any) {
      if (this.$options.router) {
        this._routerRoot = this
        this._router = this.$options.router
        this._router.init(this)
        // 使用 reactive 代理
        this._route = reactive(this._router.currentRoute.value)
      } else {
        this._routerRoot = (this.$parent && this.$parent._routerRoot) || this
      }
    }
  })

  app.config.globalProperties.$router = /* ... */
  app.config.globalProperties.$route  = /* ... */
  app.component('RouterView',  RouterView)
  app.component('RouterLink',  RouterLink)
}

              # 6. 面试 8 连问 & 一句话答案

              问题 一句话答案
              路由懒加载原理？ 利用 import() 动态 chunk，匹配到时再加载组件。
              导航守卫执行顺序？ 全局 beforeEach → 路由独享 → 组件内 → beforeResolve → afterEach
              Hash 与 History 差异？ Hash 利用 location.hash，History 利用 pushState，后者需后端配置回退。
              如何响应式路由？ Vue2 用 defineReactive(_route)；Vue3 用 reactive(currentRoute)
              如何动态添加路由？ router.addRoute(parentName, route)
              如何捕获 404？ 配置 { path: '*', component: NotFound }（Vue2）或 { path: '/:pathMatch(.*)*', ... }（Vue3）
              路由鉴权最佳实践？ 在 beforeEach 中判断 meta.requiresAuth 并检查 token / 权限。
              SSR 如何同构路由？ 服务端 router.push(url) 拿到首屏组件 → 渲染 → 客户端 hydrate。
