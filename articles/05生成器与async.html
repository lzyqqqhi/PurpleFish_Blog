    # 生成器实现async

    • 迷你调度器 - 把生成器“变成” async 函数

      function _async(genFn) {
  return function (...args) {
    return new Promise((resolve, reject) => {
      const g = genFn(...args);

      function step(val) {
        try {
          const { value, done } = g.next(val);
          if (done) return resolve(value);
    
          // 支持 Promise、普通值、数组（并行）
          const p = value instanceof Array
            ? Promise.all(value)
            : Promise.resolve(value);
          p.then(step, reject);
        } catch (err) {
          reject(err);
        }
      }
      step();
    });

  };
}

    • 用生成器写“异步函数”

      // 模拟网络请求
const sleep = ms => new Promise(r => setTimeout(r, ms));

// ↓↓↓ 这就是用生成器写的 async 函数 ↓↓↓
const fetchUser = _async(function* (id) {
  console.log('⏳ 开始请求用户', id);
  yield sleep(1000);                 // 模拟网络延迟
  if (id < 0) throw new Error('id 非法');
  const data = { id, name: 'Alice' };
  console.log('✅ 收到数据', data);
  return data;
});

// 顺序 + 错误示范
const demo = _async(function* () {
  try {
    const u1 = yield fetchUser(1);   // 顺序
    const u2 = yield fetchUser(2);
    console.log('🚀 并行开始');
    const [u3, u4] = yield [
      fetchUser(3),
      fetchUser(4)
    ];
    console.log('全部完成', [u1, u2, u3, u4]);
    yield fetchUser(-1);             // 故意抛错
  } catch (e) {
    console.error('捕获错误', e);
  }
});

      # ps

      在生成器里：

      • 只要函数 还没有执行到 return（包括隐式 return undefined）， done 就一直是 false。
      • 只有 真正 return 了某个值，下一次 next() 才会得到 { value: 返回值, done: true }。
