# 类型定义

  • JS 原生数据类型：number、string、boolean、object、undefined、null、symbol、bigint
  • TypeScript 提供静态类型校验
const name: string = 'heyi'
const age: number = 18

# 联合类型与交叉类型

  • 联合类型 (A | B): 变量可为多种类型之一

    • let myFavoriteNumber: string | number
myFavoriteNumber = 7;
myFavoriteNumber = '7';

  • 交叉类型 (A & B): 同时拥有多种类型的特性

    • type NameProtocol = { name: string }
type PersonProtocol = { age: number; say: () => void }
type Student = NameProtocol & PersonProtocol

# 泛型基础

  • 用于定义可复用、类型安全的函数、接口或类
function identity<T>(arg: T): T {
    return arg;
}

class Queue<T> {
    private data: T[] = [];
    push = (item: T) => this.data.push(item);
    pop = (): T | undefined => this.data.shift();
}

const numQueue = new Queue<number>();
numQueue.push(1);
// numQueue.push('2'); // 编译错误

# extends、keyof、typeof

  • extends：用来约束泛型参数类型
function sayHello<P extends {name: string}>(person: P) {
  console.log(person.name)
}
  • keyof/typeof：类型查询/类型映射
type Pick<T, K extends keyof T> = { [P in K]: T[P] }

# infer（类型推断）

  • 获取函数参数类型：
type ParamType<T> = T extends (arg: infer P) => any ? P : T

type User = { name: string; age: number }
type Func = (user: User) => void;
type Param = ParamType<Func>; // { name: string; age: number }
  • 获取函数返回值类型：
type ReturnType<T> = T extends (...args: any[]) => infer P ? P : any

# 类型守卫与类型保护

  • 类型守卫：将类型判断逻辑包裹，辅助类型推导
function isHTMLElement(e: EventTarget | null): e is HTMLElement {
  return e instanceof HTMLElement
}
  • 在实际编程中广泛应用于 DOM、API 类型推断。

# 装饰器和反射

  • 配置 experimentalDecorators: true 即可使用
  • 常用于 NestJS、Angular 等框架
  • 应用场景：类、属性、方法、参数等
function log(target: Object, propertyKey: string, descriptor: PropertyDescriptor) {
  console.log(`call: ${propertyKey}`);
}

class C {
  @log
  method() {}
}
  • 利用 Reflect Metadata 可为对象添加、读取元数据
