        # JavaScript异步编程

        # Promise

        # resolve 和 reject 基本原理

        • 执行 resolve，Promise 状态变为 fulfilled。
        • 执行 reject，状态变为 rejected。
        • Promise 的状态是不可逆的，只能从 pending 到 fulfilled 或 rejected，一旦变化就不可再次改变。
        • Promise 内部 throw 也会等同于调用 reject。

        # 简易Promise实现

        class MyPromise {
    constructor(executor) {
        this.PromiseState = 'pending';
        this.PromiseResult = null;
        this.resolve = this.resolve.bind(this);
        this.reject = this.reject.bind(this);
        try {
            executor(this.resolve, this.reject);
        } catch (e) {
            this.reject(e);
        }
    }
    resolve(value) {
        if (this.PromiseState !== 'pending') return;
        this.PromiseState = 'fulfilled';
        this.PromiseResult = value;
    }
    reject(reason) {
        if (this.PromiseState !== 'pending') return;
        this.PromiseState = 'rejected';
        this.PromiseResult = reason;
    }
}

        # 题 4 then 里再抛宏任务

        console.log('A');
setTimeout(() => console.log('B'), 0);
Promise.resolve()
  .then(() => {
    console.log('C');
    setTimeout(() => console.log('D'), 0);
  })
  .then(() => console.log('E'));
console.log('F');

        步骤

        1. 同步：A → F
        2. 微任务：C
          • 在 C 的回调中把 D 这个宏任务推入下一轮
        3. 微任务：E
        4. 宏任务：B （第 1 个 setTimeout）
        5. 宏任务：D （第 2 个 setTimeout）

        打印

        A F C E B D

        # 题 5 多层嵌套 + async/await

        console.log(1);
async function a() {
  console.log(2);
  await Promise.resolve();   // 把后面代码变成微任务
  console.log(3);
}
setTimeout(() => console.log(4));
a();
new Promise(res => {
  console.log(5);
  res();
}).then(() => console.log(6))
  .then(() => console.log(7));
console.log(8);

        步骤

        1. 同步：1 → 2（a 函数立即执行）→ 5（Promise 构造函数同步执行）→ 8
        2. 微任务队列（按入队顺序）：
          • await 后的部分 → 3
          • Promise 的 then → 6 → 7
        3. 宏任务：4

        打印

        1 2 5 8 3 6 7 4

        # 速记口诀（浏览器）

        每轮事件循环

        1. 清空 同步
        2. 清空 微任务（queueMicrotask / Promise.then / await 后的代码）
        3. 取一个 宏任务（setTimeout / setInterval / I/O / UI 事件）
        4. 回到 2. 直到宏任务队列空

        # 异步 & 多次 then 支持

        • 当 Promise 状态 pending 时，then 的回调需要缓存，待状态变化后执行。
        class MyPromise {
    constructor(executor) {
        this.PromiseResult = null;
        this.PromiseState = 'pending';
        this.onFulfilledCallbacks = [];
        this.onRejectedCallbacks = [];
        this.resolve = this.resolve.bind(this);
        this.reject = this.reject.bind(this);
        try {
            executor(this.resolve, this.reject);
        } catch (e) {
            this.reject(e);
        }
    }

    resolve(value) {
        if (this.PromiseState !== 'pending') return;
        this.PromiseState = 'fulfilled';
        this.PromiseResult = value;
        while (this.onFulfilledCallbacks.length) {
            this.onFulfilledCallbacks.shift()(this.PromiseResult);
        }
    }

    reject(reason) {
        if (this.PromiseState !== 'pending') return;
        this.PromiseState = 'rejected';
        this.PromiseResult = reason;
        while (this.onRejectedCallbacks.length) {
            this.onRejectedCallbacks.shift()(this.PromiseResult);
        }
    }

    then(onFulfilled, onRejected) {
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : val => val;
        onRejected = typeof onRejected === 'function' ? onRejected : reason => { throw reason };

        if (this.PromiseState === 'fulfilled') {
            onFulfilled(this.PromiseResult);
        } else if (this.PromiseState === 'rejected') {
            onRejected(this.PromiseResult);
        } else if (this.PromiseState === 'pending') {
            this.onFulfilledCallbacks.push(onFulfilled.bind(this));
            this.onRejectedCallbacks.push(onRejected.bind(this));
        }
    }
}

        # Promise衍生方法

        # all

        • 接收 Promise 数组，全部 fulfilled 返回全部结果，否则第一个 rejected 立即 rejected。
        MyPromise.all = function (promises) {
  return new MyPromise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Argument is not an array'));
    }

    const len = promises.length;
    if (len === 0) return resolve([]);

    const result = new Array(len);
    let fulfilledCount = 0;

    promises.forEach((p, i) => {
      MyPromise.resolve(p).then(
        value => {
          result[i] = value;
          if (++fulfilledCount === len) resolve(result);
        },
        reason => reject(reason)   // 只要有一个 reject，整体立即 reject
      );
    });
  });
};

        # race

        • 返回第一个改变状态的 Promise 的值。

          MyPromise.race = function (promises) {
  return new MyPromise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Argument is not an array'));
    }

    promises.forEach(p => {
      MyPromise.resolve(p).then(resolve, reject);
    });
  });
};

        # allSettled

        • 无论成功失败，返回所有结果对象数组。

          MyPromise.allSettled = function (promises) {
  return new MyPromise(resolve => {
    if (!Array.isArray(promises)) {
      return resolve([]);
    }

    const len = promises.length;
    if (len === 0) return resolve([]);

    const result = new Array(len);
    let settledCount = 0;

    promises.forEach((p, i) => {
      MyPromise.resolve(p)
        .then(
          value => {
            result[i] = { status: 'fulfilled', value };
          },
          reason => {
            result[i] = { status: 'rejected', reason };
          }
        )
        .finally(() => {
          if (++settledCount === len) resolve(result);
        });
    });
  });
};

        # any

        • 任一 Promise fulfilled 即返回，否则全部 rejected 后整体 rejected。
        MyPromise.any = function (promises) {
  return new MyPromise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Argument is not an array'));
    }

    const len = promises.length;
    if (len === 0) {
      return reject(new AggregateError([], 'All promises were rejected'));
    }

    const errors = new Array(len);
    let rejectedCount = 0;

    promises.forEach((p, i) => {
      MyPromise.resolve(p).then(
        value => resolve(value), // 只要有一个成功即可
        reason => {
          errors[i] = reason;
          if (++rejectedCount === len) {
            reject(new AggregateError(errors, 'All promises were rejected'));
          }
        }
      );
    });
  });
};

// 简易 AggregateError Polyfill（若环境没有）
if (!globalThis.AggregateError) {
  globalThis.AggregateError = class AggregateError extends Error {
    constructor(errors, message) {
      super(message);
      this.errors = errors;
      this.name = 'AggregateError';
    }
  };
}

        # all实现race

        MyPromise.raceByAll = function (promises) {
  return new MyPromise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Argument is not an array'));
    }

    // 把每个 promise 包装成“谁先落定就抛一个特殊标记”
    const wrapped = promises.map(p =>
      MyPromise.resolve(p).then(
        val => Promise.reject({ done: true, val }),
        err => Promise.reject({ done: true, err })
      )
    );

    MyPromise.all(wrapped).then(
      () => { /* 永远不会走到这里 */ },
      wrapper => {
        if (wrapper && wrapper.done) {
          wrapper.val !== undefined ? resolve(wrapper.val) : reject(wrapper.err);
        } else {
          reject(wrapper); // 真实异常
        }
      }
    );
  });
};

        # race实现any

        MyPromise.anyByRace = function (promises) {
  return new MyPromise((resolve, reject) => {
    if (!Array.isArray(promises)) {
      return reject(new TypeError('Argument is not an array'));
    }

    const len = promises.length;
    if (len === 0) {
      return reject(new AggregateError([], 'All promises were rejected'));
    }

    const errors = new Array(len);
    let remaining = len;

    // 把每个 promise 转成“成功即 resolve，失败则记录并继续”
    const wrapped = promises.map((p, i) =>
      MyPromise.resolve(p).catch(err => {
        errors[i] = err;
        if (--remaining === 0) {
          throw new AggregateError(errors, 'All promises were rejected');
        }
        // 返回一个永远挂起的 Promise，确保 race 不会选中它
        return new MyPromise(() => {});
      })
    );

    MyPromise.race(wrapped).then(resolve, reject);
  });
};

        # 带并发限制的调度器 Scheduler

        • 需求：指定并发数量，自动调度任务，顺序输出。
        class Scheduler {
  constructor(max) {
    this.max = max;
    this.count = 0; // 当前执行数
    this.queue = [];
  }
  async add(promiseCreator) {
    if (this.count >= this.max) {
      await new Promise(resolve => this.queue.push(resolve));
    }
    this.count++;
    let res = await promiseCreator();
    this.count--;
    if (this.queue.length) {
      this.queue.shift()();
    }
    return res;
  }
}
