# 手写 bind 方法

  • 完整代码
Function.prototype.bind2 = function (context) {
    if (typeof this !== "function") {
      throw new Error("Function.prototype.bind - what is trying to be bound is not callable");
    }
    var self = this;
    var args = Array.prototype.slice.call(arguments, 1);

    var fNOP = function () {};

    var fBound = function () {
        var bindArgs = Array.prototype.slice.call(arguments);
        return self.apply(this instanceof fNOP ? this : context, args.concat(bindArgs));
    }

    fNOP.prototype = this.prototype;
    fBound.prototype = new fNOP();
    return fBound;
}

# 设计模式

# 工厂模式

  • 用一个“工厂”对象/函数来负责创建其他对象，调用者只关心要什么，而不关心怎么造

  • 优点：实现简单 ，低耦合

  • 缺点：无法通过 instanceof 或构造函数判断对象类型；所有实例无法共享原型对象。

    function createPerson(name) {
    var o = new Object();
    o.name = name;
    o.getName = function () {
        console.log(this.name);
    };
    return o;
}

var person1 = createPerson('kevin');

# 构造函数模式（优化）

  • 用“构造函数 + new”来批量生产相似对象**——把公共属性和方法挂在实例或原型上，天然支持面向对象中的“类实例化”**
  • 优点：实例可识别为特定类型，可用 instanceof 判断
  • 缺点：每次创建实例方法都会新建一份（已优化）
function Person(name) {
    this.name = name;
    this.getName = getName;
}
function getName() {
    console.log(this.name);
}

# 原型模式

  • 把“共享的东西”沉淀到prototype，把“变化的东西”留在实例自身
  • 优点：方法不会被重复创建，节约内存
  • 缺点：所有属性和方法被所有实例共享，无法初始化参数
function Person() {}

Person.prototype = {
    constructor: Person,
    name: 'kevin',
    getName: function () {
        console.log(this.name);
    }
};

# 动态原型模式

function Person(name) {
    this.name = name;
    if(typeof this.getName != "function") {
        Person.prototype.getName = function () {
            console.log(this.name);
        }
    }
}

# 组合模式

  • 构造函数模式+原型模式

  • 优点：共享和私有属性、方法分离，最常用

  • 缺点：封装性稍逊，原型定义和构造函数写在不同位置

function Person(name) {
    this.name = name;
}

Person.prototype = {
    constructor: Person,
    getName: function () {
        console.log(this.name);
    }
};

# 继承方式

# 原型链继承

  • 把子类的原型对象（Child.prototype）指向父类的一个实例（new Parent()），让子类实例可以通过原型链访问到父类的属性与方法。
function Parent () {
    this.name = 'xianzao';
}
Parent.prototype.getName = function () {
    console.log(this.name);
}
function Child () {}
Child.prototype = new Parent();

var child1 = new Child();
console.log(child1.getName()) // xianzao

缺点：引用类型属性会被所有实例共享

# 拷贝原型链继承

  • 既想拿到父类原型上的方法，又不想把父类的实例属性挂到子类原型上；同时避免组合继承里“两次调用父类构造函数”的浪费。书中给出的最终手段是 “寄生组合式继承” ——本质上就是把父类原型对象浅拷贝一份，再赋给子类，而不是把整个父实例挂过去
// 父类
function Parent(name) {
  this.name = name;
}
Parent.prototype.say = () => console.log(this.name);

// 子类
function Child(name, age) {
  Parent.call(this, name);
  this.age = age;
}

// 拷贝原型链（寄生组合）
Child.prototype = Object.create(Parent.prototype);
Child.prototype.constructor = Child;

// 使用
new Child('Tom', 18).say(); // Tom

# 构造函数借用（经典继承）

  • 在子构造函数里直接调用父构造函数，并把父构造函数的 this 绑定到子实例上，从而“借用”父类为实例准备的属性/方法。
function Parent () {
    this.names = ['xianzao', 'zaoxian'];
}
function Child () {
    Parent.call(this);
}
var child1 = new Child();
child1.names.push('test');
console.log(child1.names); // ["xianzao", "zaoxian", "test"]
var child2 = new Child();
console.log(child2.names); // ["xianzao", "zaoxian"]

优点：避免引用类型属性被共享；可向父类传递参数

缺点：方法被定义在构造函数内部，每个实例都有一份

# 组合继承

  • 把 “原型链继承” 和 “构造函数借用” 两种技术组合起来，既能让子类继承父类的原型方法，又能让每个子实例拥有独立的父类实例属性，并支持向父类动态传参。
function Parent (name) {
    this.name = name;
    this.colors = ['red', 'blue', 'green'];
}
Parent.prototype.getName = function () {
    console.log(this.name)
}
function Child (name, age) {
    Parent.call(this, name); // 调用父构造函数
    this.age = age;
}
Child.prototype = new Parent(); // 原型链继承
Child.prototype.constructor = Child;

var child1 = new Child('kevin', '18');
child1.colors.push('black');
console.log(child1.colors); // ["red", "blue", "green", "black"]
var child2 = new Child('daisy', '20');
console.log(child2.colors); // ["red", "blue", "green"]

# 组合寄生式继承

  • 在组合继承基础上，用“寄生”手段只拷贝父类原型，避免再次调用父构造函数，从而解决组合继承两次调用 Parent”的冗余。
function inherit(child, parent) {
  child.prototype = Object.create(parent.prototype); // 只拷贝原型
  child.prototype.constructor = child;
}

function Parent(name) {
  this.name = name;
}
Parent.prototype.say = () => console.log(this.name);

function Child(name, age) {
  Parent.call(this, name);   // 借构造函数
  this.age = age;
}

inherit(Child, Parent);      // 寄生组合
new Child('Tom', 18).say();  // Tom

# 原型继承

  • 直接用“某个对象”作为原型来创建新对象，让新对象天然继承原型的属性和方法；JavaScript 语言级实现就是 Object.create
function createObj(o) {
    function F(){}
    F.prototype = o;
    return new F();
}
var person = {
    name: 'kevin',
    friends: ['daisy', 'kelly']
}
var person1 = createObj(person);
var person2 = createObj(person);
person1.friends.push('taylor');
console.log(person2.friends); // ["daisy", "kelly", "taylor"]
